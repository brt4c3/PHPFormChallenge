Certainly! Below are the separate implementations for the "Select All" checkbox functionality and the individual checkbox selection functionality, along with a separate function to handle individual checkbox selection.

### Individual Checkbox Selection Functionality:

```javascript
// Function to handle individual checkbox selection
function handleCheckboxSelection() {
    let otherCheckboxes = document.querySelectorAll('.check-industry');
    for (let checkbox of otherCheckboxes) {
        checkbox.addEventListener('change', function() {
            let allChecked = true;
            for (let checkbox of otherCheckboxes) {
                if (!checkbox.checked) {
                    allChecked = false;
                    break;
                }
            }
            document.getElementById('industry_checkAll').checked = allChecked;
        });
    }
}

// Call the function to set up individual checkbox selection behavior
handleCheckboxSelection();
```

### Select All Checkbox Functionality:

```javascript
// Function to handle the "Select All" checkbox
function handleSelectAllCheckbox() {
    let selectAllCheckbox = document.getElementById('industry_checkAll');
    let otherCheckboxes = document.querySelectorAll('.check-industry');
    selectAllCheckbox.addEventListener('change', function() {
        for (let checkbox of otherCheckboxes) {
            checkbox.checked = this.checked;
        }
    });
}

// Call the function to set up "Select All" checkbox behavior
handleSelectAllCheckbox();
```

These separate functions ensure that the "Select All" checkbox and the individual checkbox selection functionality are decoupled, allowing them to be implemented and managed independently. You can call these functions wherever necessary to set up the desired checkbox behavior.
